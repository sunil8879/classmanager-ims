-- Helper: Get current user's Institute ID
CREATE OR REPLACE FUNCTION public.get_my_institute_id() 
RETURNS text AS $$
  SELECT institute_id FROM public.profiles WHERE id = auth.uid();
$$ LANGUAGE sql SECURITY DEFINER;

-- Helper: Get current user's Role
CREATE OR REPLACE FUNCTION public.get_my_role() 
RETURNS text AS $$
  SELECT role FROM public.profiles WHERE id = auth.uid();
$$ LANGUAGE sql STABLE SECURITY DEFINER;

-- Logic: Auto-sync payments from Student UI to Payment History
CREATE OR REPLACE FUNCTION public.auto_sync_payment() 
RETURNS trigger AS $$
BEGIN
    IF (TG_OP = 'INSERT' AND NEW.paid_amount > 0) OR 
       (TG_OP = 'UPDATE' AND NEW.paid_amount <> OLD.paid_amount) THEN
       DECLARE
           new_cash DECIMAL := CASE 
               WHEN TG_OP = 'INSERT' THEN NEW.paid_amount 
               ELSE NEW.paid_amount - OLD.paid_amount 
           END;
       BEGIN
           IF new_cash > 0 THEN
               INSERT INTO payment_history (institute_id, student_id, roll_number, amount, payment_date, month_year)
               VALUES (NEW.institute_id, NEW.id, NEW.roll_number, new_cash, CURRENT_DATE, trim(to_char(CURRENT_DATE, 'FMMonth YYYY')));
           END IF;
       END;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- ========================================================
-- 3. CREATE RLS POLICIES FOR MULTI-TENANCY (By Institute ID)
-- ========================================================

-- Allow users to VIEW records only for their own institute
CREATE POLICY "View leaves for own institute" 
ON public.leave_management FOR SELECT 
USING (institute_id = (SELECT institute_id FROM public.profiles WHERE id = auth.uid()));

-- Allow users to INSERT records only for their own institute
CREATE POLICY "Insert leaves for own institute" 
ON public.leave_management FOR INSERT 
WITH CHECK (institute_id = (SELECT institute_id FROM public.profiles WHERE id = auth.uid()));

-- Allow users to UPDATE records only for their own institute
CREATE POLICY "Update leaves for own institute" 
ON public.leave_management FOR UPDATE 
USING (institute_id = (SELECT institute_id FROM public.profiles WHERE id = auth.uid()));

-- Allow users to DELETE records only for their own institute
CREATE POLICY "Delete leaves for own institute" 
ON public.leave_management FOR DELETE 
USING (institute_id = (SELECT institute_id FROM public.profiles WHERE id = auth.uid()));
